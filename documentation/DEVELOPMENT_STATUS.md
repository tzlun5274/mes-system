```bash
# 測試環境啟動
cd /var/www/mes_test
./start_test_server.sh

# 測試環境 Celery
cd /var/www/mes_test
./start_test_celery.sh

# 建立預設定時任務
python3 manage.py setup_auto_approval_tasks

# 檢查系統狀態
python3 manage.py check

# 資料庫遷移
python3 manage.py migrate

# 收集靜態檔案
python3 manage.py collectstatic
```
